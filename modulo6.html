function gerarCertificado(){ let nome = localStorage.getItem("nomeAluno") || "Aluno"; let turma = localStorage.getItem("turmaAluno") || "Turma"; let escola = localStorage.getItem("escola") || "Não definida"; let data = new Date().toLocaleDateString("pt-BR"); let codigo = "PEPI-" + Math.floor(10000 + Math.random() * 90000); let user = firebase.auth().currentUser; if(!user){ alert("Faça login como gestor"); return; } db.collection("certificados").doc(codigo).set({ nome: nome, turma: turma, data: data, gestorId: user.uid, escola: escola }); alert("Certificado gerado com sucesso!"); }
