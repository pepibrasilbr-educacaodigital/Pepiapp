function salvarCertificado(){ const user = firebase.auth().currentUser; if(!user){ alert("Você precisa estar logado"); return; } let nome = document.getElementById("nome").value; let turma = document.getElementById("turma").value; let escola = localStorage.getItem("escola"); let codigo = "PEPI-" + Math.floor(Math.random() * 100000); let data = new Date().toLocaleDateString(); db.collection("certificados").doc(codigo).set({ nome: nome, turma: turma, escola: escola, data: data, // 🔥 ESSENCIAL gestorId: user.uid }) .then(() => { alert("Certificado gerado com sucesso!"); }) .catch(error => { console.error(error); alert("Erro ao salvar"); }); }
